<script setup>
import { onMounted } from 'vue'

onMounted(() => {
  const btn = document.getElementById('copy-page-btn')
  if (!btn) return
  btn.addEventListener('click', () => {
    const content = document.querySelector('.vp-doc') || document.querySelector('.VPPage') || document.querySelector('main')
    if (!content) return

    function htmlToMd(el) {
      let md = ''
      function walk(node) {
        if (node.nodeType === 3) { md += node.textContent; return }
        if (node.nodeType !== 1) return
        const tag = node.tagName.toLowerCase()
        if (node.classList && (node.classList.contains('copy-bar-btn') || node.classList.contains('copy-bar'))) return
        if (['style','script','nav','aside'].includes(tag)) return
        if (tag === 'h1') { md += '\n# '; node.childNodes.forEach(walk); md += '\n\n'; return }
        if (tag === 'h2') { md += '\n## '; node.childNodes.forEach(walk); md += '\n\n'; return }
        if (tag === 'h3') { md += '\n### '; node.childNodes.forEach(walk); md += '\n\n'; return }
        if (tag === 'h4') { md += '\n#### '; node.childNodes.forEach(walk); md += '\n\n'; return }
        if (tag === 'p') { node.childNodes.forEach(walk); md += '\n\n'; return }
        if (tag === 'br') { md += '\n'; return }
        if (tag === 'strong' || tag === 'b') { md += '**'; node.childNodes.forEach(walk); md += '**'; return }
        if (tag === 'em' || tag === 'i') { md += '*'; node.childNodes.forEach(walk); md += '*'; return }
        if (tag === 'code' && node.parentElement.tagName.toLowerCase() !== 'pre') {
          md += '`' + node.textContent + '`'; return
        }
        if (tag === 'pre') { md += '\n```\n' + node.textContent + '\n```\n\n'; return }
        if (tag === 'a') { md += '['; node.childNodes.forEach(walk); md += '](' + (node.href || '') + ')'; return }
        if (tag === 'li') { md += '- '; node.childNodes.forEach(walk); md += '\n'; return }
        if (tag === 'ul' || tag === 'ol') { md += '\n'; node.childNodes.forEach(walk); md += '\n'; return }
        if (tag === 'table') {
          const rows = node.querySelectorAll('tr')
          rows.forEach((row, i) => {
            const cells = row.querySelectorAll('th, td')
            md += '| ' + Array.from(cells).map(c => c.textContent.trim()).join(' | ') + ' |\n'
            if (i === 0) md += '|' + Array.from(cells).map(() => '---').join('|') + '|\n'
          })
          md += '\n'
          return
        }
        if (tag === 'hr') { md += '\n---\n\n'; return }
        if (tag === 'blockquote') { md += '> '; node.childNodes.forEach(walk); md += '\n'; return }
        if (tag === 'img') return
        node.childNodes.forEach(walk)
      }
      walk(el)
      return md.replace(/\n{3,}/g, '\n\n').trim()
    }

    const md = htmlToMd(content)
    navigator.clipboard.writeText(md).then(() => {
      btn.classList.add('copied')
      const origHTML = btn.innerHTML
      btn.innerHTML = '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="20 6 9 17 4 12"/></svg> Copiado!'
      setTimeout(() => {
        btn.classList.remove('copied')
        btn.innerHTML = origHTML
      }, 3000)
    }).catch(() => {
      // Fallback para contextos sem clipboard API
      const ta = document.createElement('textarea')
      ta.value = md
      document.body.appendChild(ta)
      ta.select()
      document.execCommand('copy')
      document.body.removeChild(ta)
      btn.classList.add('copied')
      const origHTML = btn.innerHTML
      btn.innerHTML = '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="20 6 9 17 4 12"/></svg> Copiado!'
      setTimeout(() => {
        btn.classList.remove('copied')
        btn.innerHTML = origHTML
      }, 3000)
    })
  })
})
</script>


## Identidade Fundamental
O **Grupo CSV (Cuidados em Saúde com Valor)** é um ecossistema estratégico de orquestração em saúde que integra gestão, educação e tecnologia para entregar valor sustentável.

Não se trata de uma holding tradicional, mas de uma **infraestrutura relacional** que conecta stakeholders dispersos (operadoras, hospitais, profissionais e pacientes) em torno de desfechos clínicos pertinentes e eficiência econômica.

## Princípios Norteadores
1.  **Valor em Saúde (VBHC):** A equação central é Desfecho / Custo. Todas as iniciativas devem melhorar o numerador (saúde) ou otimizar o denominador (recursos), preferencialmente ambos.
2.  **Orquestração Multistakeholder:** O Grupo CSV não atua em silos. Ele coordena a interação entre pagadores, prestadores e pacientes para reduzir atritos e alinhar incentivos.
3.  **Integração Sistêmica:** A solução nunca é apenas tecnológica, apenas gerencial ou apenas educacional. É a combinação das três.

## Estrutura de Atuação (A Tríade)
O Grupo opera através de três braços especializados e interdependentes:

| Braço | Função Sistêmica | Foco de Atuação | Entregável Principal |
| :--- | :--- | :--- | :--- |
| **AxiaCare®** | **Gestão & Estratégia** | Consultoria, Governança Clínica, Operação | Modelos de Negócio, Eficiência Operacional, Compliance |
| **MedValor®** | **Educação & Cultura** | Formação de Lideranças, Mudança de Mindset | Cursos, Mentorias, Capacitação Híbrida (Médico-Gestor) |
| **TheraTech®** | **Tecnologia & IA** | Desenvolvimento de Software, Inteligência de Dados | Plataformas SaaS, Algoritmos de Decisão, Automação |

## Posicionamento de Mercado
O Grupo CSV atua no modelo **B2B2X**: serve organizações de saúde (B2B) para impactar a vida dos pacientes e profissionais (B2C/B2P).

A proposta de valor é transformar a complexidade fragmentada do setor em sistemas de decisão estruturados, éticos e executáveis.
